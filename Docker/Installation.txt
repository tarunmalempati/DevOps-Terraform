1. Update your system

sudo apt update
sudo apt upgrade

2. Install pre-requisite packages

sudo apt-get install curl apt-transport-https ca-certificates software-properties-common

3. Add docker repositories 

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"
sudo apt update
apt-cache policy docker-ce

4. Install docker on Ubuntu

sudo apt install docker-ce

5.Check docker status

systemctl status docker
docker --version
                                                    
                                            OR

1. Update Repositories
  sudo apt-get update
  
2. Remove prior installations
    sudo apt-get remove docker docker-engine docker.io
    
3. Install Docker
    apt-get install docker.io
    
4. Start and automate Docker
    sudo systemctl start docker
    sudo systemctl enable docker
    
5. Check version
    docker --version

                                                                 INSTALL DOCKER COMPOSE

1.     sudo curl -L https://github.com/docker/compose/releases/download/1.29.2/docker-compose-`uname -s`-`uname -m` -o /usr/local/bin/docker-compose
2.     sudo chmod +x /usr/local/bin/docker-compose
3.     docker-compose --version

                                                                  INSTALL DOCKER SCOUT  
1. mkdir -p ~/.docker/cli-plugins  
2. wget wget https://github.com/docker/scout-cli/releases/download/v0.15.0/docker-scout_0.15.0_linux_amd64.tar.gz  
3. tar -xvzf docker-scout_0.15.0_linux_amd64.tar.gz  
4. mv docker-scout ~/.docker/cli-plugins  
5. chmod +x ~/.docker/cli-plugins/docker-scout  

                                                                 INSTALL DOCKER BUILDX
1. Install docker using the above given commands.
2. mkdir -p ~/.docker/cli-plugins/
3. curl -SL https://github.com/docker/buildx/releases/download/v0.10.5/buildx-v0.10.5.linux-amd64 -o ~/.docker/cli-plugins/docker-buildx
4. chmod +x ~/.docker/cli-plugins/docker-buildx
5. docker buildx version
